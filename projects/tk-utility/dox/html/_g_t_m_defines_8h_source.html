00039 // _GTMDevLog & _GTMDevAssert
00040 //
00041 // _GTMDevLog & _GTMDevAssert are meant to be a very lightweight shell for
00042 // developer level errors.  This implementation simply macros to NSLog/NSAssert.
00043 // It is not intended to be a general logging/reporting system.
00044 //
00045 // Please see http://code.google.com/p/google-toolbox-for-mac/wiki/DevLogNAssert
00046 // for a little more background on the usage of these macros.
00047 //
00048 //    _GTMDevLog           log some error/problem in debug builds
00049 //    _GTMDevAssert        assert if conditon isn't met w/in a method/function
00050 //                           in all builds.
00051 // 
00052 // To replace this system, just provide different macro definitions in your
00053 // prefix header.  Remember, any implementation you provide *must* be thread
00054 // safe since this could be called by anything in what ever situtation it has
00055 // been placed in.
00056 // 
00057 
00058 // We only define the simple macros if nothing else has defined this.
00059 #ifndef _GTMDevLog
00060 
00061 #ifdef DEBUG
00062  #define _GTMDevLog(...) NSLog(__VA_ARGS__)
00063 #else
00064  #define _GTMDevLog(...) do { } while (0)
00065 #endif
00066 
00067 #endif // _GTMDevLog

00074 #ifndef _GTMDevAssert
00075 // we directly invoke the NSAssert handler so we can pass on the varargs
00076 // (NSAssert doesn't have a macro we can use that takes varargs)
00077 #if !defined(NS_BLOCK_ASSERTIONS)
00078 #define _GTMDevAssert(condition, ...)                                    \
00079   do {                                                                   \
00080     if (!(condition)) {                                                  \
00081       [[NSAssertionHandler currentHandler]                               \
00082           handleFailureInFunction:[NSString stringWithCString:__PRETTY_FUNCTION__] \
00083                              file:[NSString stringWithCString:__FILE__]  \
00084                        lineNumber:__LINE__                               \
00085                       description:__VA_ARGS__];                          \
00086     }                                                                    \
00087   } while(0)
00088 #else // !defined(NS_BLOCK_ASSERTIONS)
00089 #define _GTMDevAssert(condition, ...) do { } while (0)
00090 #endif // !defined(NS_BLOCK_ASSERTIONS)
00091 
00092 #endif // _GTMDevAssert

00094 // _GTMCompileAssert
00095 // _GTMCompileAssert is an assert that is meant to fire at compile time if you
00096 // want to check things at compile instead of runtime. For example if you
00097 // want to check that a wchar is 4 bytes instead of 2 you would use
00098 // _GTMCompileAssert(sizeof(wchar_t) == 4, wchar_t_is_4_bytes_on_OS_X)
00099 // Note that the second "arg" is not in quotes, and must be a valid processor
00100 // symbol in it's own right (no spaces, punctuation etc).
00101 
00102 // Wrapping this in an #ifndef allows external groups to define their own
00103 // compile time assert scheme.
00104 #ifndef _GTMCompileAssert
00105 // We got this technique from here:
00106 // http://unixjunkie.blogspot.com/2007/10/better-compile-time-asserts_29.html
00107 
00108 #define _GTMCompileAssertSymbolInner(line, msg) _GTMCOMPILEASSERT ## line ## __ ## msg
00109 #define _GTMCompileAssertSymbol(line, msg) _GTMCompileAssertSymbolInner(line, msg)
00110 #define _GTMCompileAssert(test, msg) \
00111   typedef char _GTMCompileAssertSymbol(__LINE__, msg) [ ((test) ? 1 : -1) ]
00112 #endif // _GTMCompileAssert
